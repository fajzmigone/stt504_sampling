\[
\hat{Y}
  = \frac{1}{n}\sum_{k \in S} y_k
  = \frac{2907}{100}
  = 29.07 \text{ ha}.
\]

\[
\hat{S}_{y}^{2}
  = \frac{n}{n-1}
    \left(
      \frac{1}{n}\sum_{k \in S} y_k^{2}
      - \hat{Y}^{\,2}
    \right)
  = \frac{100}{99}
    \left(
      \frac{154\,593}{100}
      - 29.07^{2}
    \right)
  = 707.945.
\]
